INSTALL ONEWAVE AI SKILLS (business essentials):
────────────────────────────────────────────────────────────────────────────
  $ mkdir -p ~/.claude/skills
  $ unzip screenshot-to-code.zip
  $ cp -r screenshot-to-code ~/.claude/skills/

INSTALL ANTHROPIC SKILLS (creative & technical):
────────────────────────────────────────────────────────────────────────────
  $ cp -r official-anthropic-skills/algorithmic-art ~/.claude/skills/
  $ cp -r official-anthropic-skills/docx ~/.claude/skills/

INSTALL ALL AT ONCE:
────────────────────────────────────────────────────────────────────────────
  # OneWave AI skills
  $ for file in *.zip; do
      unzip -q "$file"
      cp -r "${file%.zip}" ~/.claude/skills/
    done

  # Official Anthropic skills
  $ cp -r official-anthropic-skills/* ~/.claude/skills/

  # Verify
  $ ls ~/.claude/skills/
